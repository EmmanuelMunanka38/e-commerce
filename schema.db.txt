model User {
  id        String   @id @default(cuid())
  email     String   @unique
  name      String?
  role      Role     @default(USER)
  products  Product[]
  orders    Order[]
  createdAt DateTime @default(now())
}
model Product {
  id          String   @id @default(cuid())
  title       String
  description String?
  price       Float
  images      String[]
  vendorId    String
  vendor      User     @relation(fields: [vendorId], references: [id])
  orders      OrderItem[]
  createdAt   DateTime @default(now())
}
model Order {
  id        String     @id @default(cuid())
  userId    String
  user      User       @relation(fields: [userId], references: [id])
  items     OrderItem[]
  total     Float
  status    OrderStatus @default(PENDING)
  createdAt DateTime   @default(now())
}
